                 # установка положения сервопривода №1

0 get_position   # записали в стак[0] значение из 0 канала контроля (0-1023) : (0-5 В) > (0-960) : (0-4692 мВ)

961 LESS_THAN    # если [1] меньше 961

  endif            # конец условия №3

  DROP             # стерли знаечение из [0]
